Table users {
  id integer [primary key]
  username varchar
  first_name varchar
  middle_name varchar
  last_name varchar
  email varchar
  password_hash varchar
  phone_number varchar
  avatar_url varchar [null]
  created_at timestamp
  updated_at timestamp
}

Table properties {
  id integer [primary key]
  user_id integer [ref: > users.id]
  amenity_id integer
  address varchar
  city varchar
  state varchar
  country varchar
  zipcode varchar
  name varchar
  description text
  price varchar	
  bedrooms integer	
  baths integer
  max_guests integer	
  cancellation_policy text
  coordinates_latitude varchar
  coordinates_longitude varchar
  created_at timestamp	
  updated_at timestamp
}

Table bookings {
  id	integer	[primary key]
  user_id	integer [ref: > users.id]
  property_id	integer	[ref: > properties.id]
  start_date	datetime	
  end_date	datetime	
  is_refundable boolean
  created_at	timestamp	
  updated_at	timestamp
}

Table property_images {
  id	integer	[primary key]
  property_id	integer [ref: > properties.id]
  url	varchar	
}

Table property_rules {
  id integer [primary key]
  property_id integer [ref: > properties.id]
  rule text
  isActive boolean
}

Table property_saftey_notes {
  id integer [primary key]
  property_id integer [ref: > properties.id]
  notes text
  isActive boolean
}

Table property_bed_info {
  id integer [primary key]
  property_id integer [ref: > properties.id]
  room varchar
  bedType vachar
  isActive boolean
}


Table reviews {
  id integer [primary key]
  user_id integer [ref: > users.id]
  property_id integer [ref: > properties.id]
  review text
  rating integer
  created_at	timestamp	
  updated_at	timestamp
}

Table amenities {
  id integer [primary key]
  name text
  isActive boolean
}

Table property_amenity_mapping {
  id integer [primary key]
  property_id integer [ref :> properties.id]
  amenity_id integer [ref :> amenities.id]
}

Table hosts {
  id integer [primary key]
  user_id integer [ref: > users.id]
  bio text
  isActive boolean
}
